New Tables
-- Companies
CREATE TABLE companies (
  id UUID PRIMARY KEY,
  name TEXT,
  domain TEXT,
  plan TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Company Jobs
CREATE TABLE company_jobs (
  id UUID PRIMARY KEY,
  company_id UUID REFERENCES companies(id),
  title TEXT,
  jd_text TEXT,
  jd_url TEXT,
  rolekit_ref TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Company Candidates
CREATE TABLE company_candidates (
  id UUID PRIMARY KEY,
  job_id UUID REFERENCES company_jobs(id),
  user_id UUID REFERENCES users(id),
  interview_session_id UUID,
  score_overall INT,
  submitted_at TIMESTAMP DEFAULT NOW()
);

-- Payments
CREATE TABLE payments (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  provider TEXT,
  provider_ref TEXT,
  amount INT,
  currency TEXT,
  status TEXT,
  meta JSONB,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Subscriptions
CREATE TABLE subscriptions (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  plan_type TEXT,
  status TEXT,
  role_context JSONB,
  started_at TIMESTAMP DEFAULT NOW(),
  expires_at TIMESTAMP
);

import Razorpay from "razorpay";
export const razorpay = new Razorpay({
  key_id: process.env.RAZORPAY_KEY_ID!,
  key_secret: process.env.RAZORPAY_KEY_SECRET!,
});

// Create order
POST /api/payments/initiate
router.post("/payments/initiate", async (req, res) => {
  const { planType, amount } = req.body;
  const order = await razorpay.orders.create({
    amount: amount * 100,
    currency: "INR",
    receipt: `hiready_${req.user.id}_${Date.now()}`
  });
  await db.insert("payments").values({
    user_id: req.user.id,
    provider: "razorpay",
    provider_ref: order.id,
    amount, currency: "INR", status: "created"
  });
  res.json({ orderId: order.id, key: process.env.RAZORPAY_KEY_ID });
});

// Webhook
POST /api/payments/webhook
router.post("/payments/webhook", async (req, res) => {
  const { event, payload } = req.body;
  if (event === "payment.captured") {
    const payment = payload.payment.entity;
    await db.update("payments").set({ status: "captured" }).where({ provider_ref: payment.order_id });
    await activateSubscription(payment.notes.user_id, payment.notes.plan_type);
  }
  res.status(200).json({ received: true });
});
